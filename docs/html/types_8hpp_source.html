57 struct type_descriptor
58 {
59 char type{'\0'};
60 std::string_view contained{};
61
62 std::string string() const
63 {
64 std::string s;
65 bool noContain = false;
66 if (contained.size() == 0 || (contained.size() == 1 && contained.front() == '\0'))
67 {
68 noContain = true;
69 }
70 if (type > 32)
71 {
72 s.resize(1);
73 s[0] = type;
74 }
75 else
76 {
77 auto tval = static_cast <int> (type);
78 if (tval < 0)
79 tval += 255;
80 std::stringstream sstr;
81 sstr << "\\x" << std::hex << std::setw(2) << std::setfill('0') << tval;
82 s = sstr.str();
83 }
84 if (noContain)
85 s += "\\x00";
86 else
87 s += contained.data();
88 return s;
89 }
90 };

98 template <typename FunctionT>
99 bool for_signature_do_noexcept (type_descriptor descr, FunctionT func)
100 {
101 using namespace std::string_literals;
102
103 switch (descr.type)
104 {
105 default: return false;
106 case('y'): func(uint8_t{}); break;
107 case('b'): func(bool{}); break;
108 case('n'): func(int16_t{}); break;
109 case('q'): func(uint16_t{}); break;
110 case('i'): func(int32_t{}); break;
111 case('u'): func(uint32_t{}); break;
112 case('x'): func(int64_t{}); break;
113 case('t'): func(uint64_t{}); break;
114 case('d'): func(double{}); break;
115 case('s'): func(std::string{}); break;
116 case('o'): func(object_path{}); break;
117 case('h'): func(file_descriptor{}); break;
118 case('g'): func(signature{}); break;
119 }
120 return true;
121 }

123 template <typename FunctionT>
124 void for_signature_do (type_descriptor descr, FunctionT func)
125 {
126 using namespace std::string_literals;
127
128 bool suc = for_signature_do_noexcept(descr, [](auto){});
129 if (suc)
130 for_signature_do_noexcept(descr, func);
131 else
132 {
133 throw std::domain_error(
134 "resolvable_variants cannot contain arrays or other complex types"s + descr.string()
135 );
